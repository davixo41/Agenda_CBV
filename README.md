<code>python -m venv venv</code>
<br>
<code>source ./bin/scripts/activate</code>
<br>
<code>pip install -r requirments.txt</code>
<br>
<code>python manage.py makemigrations</code>
<br>
<code>python manage.py migrate</code>
<br>
